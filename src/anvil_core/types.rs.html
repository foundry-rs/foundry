8/// Represents the result of `eth_getWork`
9/// This may or may not include the block number
10#[derive(Debug, Default, PartialEq, Eq)]
11pub struct Work {
12    pub pow_hash: B256,
13    pub seed_hash: B256,
14    pub target: B256,
15    pub number: Option<u64>,
16}
17
18#[cfg(feature = "serde")]
19impl serde::Serialize for Work {
20    fn serialize<S>(&self, s: S) -> Result<S::Ok, S::Error>
21    where
22        S: Serializer,
23    {
24        if let Some(num) = self.number {
25            (&self.pow_hash, &self.seed_hash, &self.target, U256::from(num)).serialize(s)
26        } else {
27            (&self.pow_hash, &self.seed_hash, &self.target).serialize(s)
28        }
29    }
30}
