141    /// Runs the test.
142    pub fn run(&self) {
143        // Skip fork tests if the RPC url is not set.
144        if self.fork_block.is_some() && std::env::var_os("ETH_RPC_URL").is_none() {
145            eprintln!("ETH_RPC_URL is not set; skipping");
146            return;
147        }
148
149        let (prj, mut test_cmd) = setup_forge(self.name, self.style.clone());
150
151        // Wipe the default structure.
152        prj.wipe();
153
154        // Clone the external repository.
155        let repo_url = format!("https://github.com/{}/{}.git", self.org, self.name);
156        let root = prj.root().to_str().unwrap();
157        clone_remote(&repo_url, root);
158
159        // Checkout the revision.
160        if self.rev.is_empty() {
161            let mut git = Command::new("git");
162            git.current_dir(root).args(["log", "-n", "1"]);
163            println!("$ {git:?}");
164            let output = git.output().unwrap();
165            if !output.status.success() {
166                panic!("git log failed: {output:?}");
167            }
168            let stdout = String::from_utf8(output.stdout).unwrap();
169            let commit = stdout.lines().next().unwrap().split_whitespace().nth(1).unwrap();
170            panic!("pin to latest commit: {commit}");
171        } else {
172            let mut git = Command::new("git");
173            git.current_dir(root).args(["checkout", self.rev]);
174            println!("$ {git:?}");
175            let status = git.status().unwrap();
176            if !status.success() {
177                panic!("git checkout failed: {status}");
178            }
179        }
180
181        // Run installation command.
182        for install_command in &self.install_commands {
183            let mut install_cmd = Command::new(&install_command[0]);
184            install_cmd.args(&install_command[1..]).current_dir(root);
185            println!("cd {root}; {install_cmd:?}");
186            match install_cmd.status() {
187                Ok(s) => {
188                    println!("\n\n{install_cmd:?}: {s}");
189                    if s.success() {
190                        break;
191                    }
192                }
193                Err(e) => {
194                    eprintln!("\n\n{install_cmd:?}: {e}");
195                }
196            }
197        }
198
199        // Run the tests.
200        test_cmd.arg("test");
201        test_cmd.args(&self.args);
202        test_cmd.args(["--fuzz-runs=32", "--ffi", "-vvv"]);
203
204        test_cmd.envs(self.envs.iter().map(|(k, v)| (k, v)));
205        if let Some(fork_block) = self.fork_block {
206            test_cmd.env("FOUNDRY_ETH_RPC_URL", crate::rpc::next_http_archive_rpc_url());
207            test_cmd.env("FOUNDRY_FORK_BLOCK_NUMBER", fork_block.to_string());
208        }
209        test_cmd.env("FOUNDRY_INVARIANT_DEPTH", "15");
210        test_cmd.env("FOUNDRY_ALLOW_INTERNAL_EXPECT_REVERT", "true");
211
212        test_cmd.assert_success();
213    }

216/// Initializes a project with `forge init` at the given path.
217///
218/// This should be called after an empty project is created like in
219/// [some of this crate's macros](crate::forgetest_init).
220///
221/// ## Note
222///
223/// This doesn't always run `forge init`, instead opting to copy an already-initialized template
224/// project from a global template path. This is done to speed up tests.
225///
226/// This used to use a `static` `Lazy`, but this approach does not with `cargo-nextest` because it
227/// runs each test in a separate process. Instead, we use a global lock file to ensure that only one
228/// test can initialize the template at a time.
229#[allow(clippy::disallowed_macros)]
230pub fn initialize(target: &Path) {
231    println!("initializing {}", target.display());
232
233    let tpath = TEMPLATE_PATH.as_path();
234    pretty_err(tpath, fs::create_dir_all(tpath));
235
236    // Initialize the global template if necessary.
237    let mut lock = crate::fd_lock::new_lock(TEMPLATE_LOCK.as_path());
238    let mut _read = Some(lock.read().unwrap());
239    if fs::read(&*TEMPLATE_LOCK).unwrap() != b"1" {
240        // We are the first to acquire the lock:
241        // - initialize a new empty temp project;
242        // - run `forge init`;
243        // - run `forge build`;
244        // - copy it over to the global template;
245        // Ideally we would be able to initialize a temp project directly in the global template,
246        // but `TempProject` does not currently allow this: https://github.com/foundry-rs/compilers/issues/22
247
248        // Release the read lock and acquire a write lock, initializing the lock file.
249        _read = None;
250
251        let mut write = lock.write().unwrap();
252
253        let mut data = String::new();
254        write.read_to_string(&mut data).unwrap();
255
256        if data != "1" {
257            // Initialize and build.
258            let (prj, mut cmd) = setup_forge("template", foundry_compilers::PathStyle::Dapptools);
259            println!("- initializing template dir in {}", prj.root().display());
260
261            cmd.args(["init", "--force"]).assert_success();
262            prj.write_config(Config {
263                solc: Some(foundry_config::SolcReq::Version(SOLC_VERSION.parse().unwrap())),
264                ..Default::default()
265            });
266
267            // Checkout forge-std.
268            let output = Command::new("git")
269                .current_dir(prj.root().join("lib/forge-std"))
270                .args(["checkout", FORGE_STD_REVISION])
271                .output()
272                .expect("failed to checkout forge-std");
273            assert!(output.status.success(), "{output:#?}");
274
275            // Build the project.
276            cmd.forge_fuse().arg("build").assert_success();
277
278            // Remove the existing template, if any.
279            let _ = fs::remove_dir_all(tpath);
280
281            // Copy the template to the global template path.
282            pretty_err(tpath, copy_dir(prj.root(), tpath));
283
284            // Update lockfile to mark that template is initialized.
285            write.set_len(0).unwrap();
286            write.seek(std::io::SeekFrom::Start(0)).unwrap();
287            write.write_all(b"1").unwrap();
288        }
289
290        // Release the write lock and acquire a new read lock.
291        drop(write);
292        _read = Some(lock.read().unwrap());
293    }
294
295    println!("- copying template dir from {}", tpath.display());
296    pretty_err(target, fs::create_dir_all(target));
297    pretty_err(target, copy_dir(tpath, target));
298}

300/// Clones a remote repository into the specified directory. Panics if the command fails.
301pub fn clone_remote(repo_url: &str, target_dir: &str) {
302    let mut cmd = Command::new("git");
303    cmd.args(["clone", "--no-tags", "--recursive", "--shallow-submodules"]);
304    cmd.args([repo_url, target_dir]);
305    println!("{cmd:?}");
306    let status = cmd.status().unwrap();
307    if !status.success() {
308        panic!("git clone failed: {status}");
309    }
310    println!();
311}

372/// Setups a new local forge project by cloning and initializing the `RemoteProject`
373///
374/// This will
375///   1. clone the prj, like "transmissions1/solmate"
376///   2. run `forge build`, if configured
377///   3. run additional commands
378///
379/// # Panics
380///
381/// If anything goes wrong during, checkout, build, or other commands are unsuccessful
382pub fn setup_forge_remote(prj: impl Into<RemoteProject>) -> (TestProject, TestCommand) {
383    try_setup_forge_remote(prj).unwrap()
384}
385
386/// Same as `setup_forge_remote` but not panicking
387pub fn try_setup_forge_remote(
388    config: impl Into<RemoteProject>,
389) -> Result<(TestProject, TestCommand)> {
390    let config = config.into();
391    let mut tmp = TempProject::checkout(&config.id).wrap_err("failed to checkout project")?;
392    tmp.project_mut().paths = config.path_style.paths(tmp.root())?;
393
394    let prj = TestProject::with_project(tmp);
395    if config.run_build {
396        let mut cmd = prj.forge_command();
397        cmd.arg("build").assert_success();
398    }
399    for addon in config.run_commands {
400        debug_assert!(!addon.is_empty());
401        let mut cmd = Command::new(&addon[0]);
402        if addon.len() > 1 {
403            cmd.args(&addon[1..]);
404        }
405        let status = cmd
406            .current_dir(prj.root())
407            .stdout(Stdio::null())
408            .stderr(Stdio::null())
409            .status()
410            .wrap_err_with(|| format!("Failed to execute {addon:?}"))?;
411        eyre::ensure!(status.success(), "Failed to execute command {:?}", addon);
412    }
413
414    let cmd = prj.forge_command();
415    Ok((prj, cmd))
416}

773    cmd: Command,
774    // initial: Command,
775    current_dir_lock: Option<parking_lot::lock_api::MutexGuard<'static, parking_lot::RawMutex, ()>>,
776    stdin_fun: Option<Box<dyn FnOnce(ChildStdin)>>,
777    /// If true, command output is redacted.
778    redact_output: bool,

978    /// Executes command, applies stdin function and returns output
979    #[track_caller]
980    pub fn execute(&mut self) -> Output {
981        self.try_execute().unwrap()
982    }
983
984    #[track_caller]
985    pub fn try_execute(&mut self) -> std::io::Result<Output> {
986        println!("executing {:?}", self.cmd);
987        let mut child =
988            self.cmd.stdout(Stdio::piped()).stderr(Stdio::piped()).stdin(Stdio::piped()).spawn()?;
989        if let Some(fun) = self.stdin_fun.take() {
990            fun(child.stdin.take().unwrap());
991        }
992        child.wait_with_output()
993    }

1002fn test_redactions() -> snapbox::Redactions {
1003    static REDACTIONS: LazyLock<snapbox::Redactions> = LazyLock::new(|| {
1004        let mut r = snapbox::Redactions::new();
1005        let redactions = [
1006            ("[SOLC_VERSION]", r"Solc( version)? \d+.\d+.\d+"),
1007            ("[ELAPSED]", r"(finished )?in \d+(\.\d+)?\w?s( \(.*?s CPU time\))?"),
1008            ("[GAS]", r"[Gg]as( used)?: \d+"),
1009            ("[AVG_GAS]", r": \d+, ~: \d+"),
1010            ("[FILE]", r"-->.*\.sol"),
1011            ("[FILE]", r"Location(.|\n)*\.rs(.|\n)*Backtrace"),
1012            ("[COMPILING_FILES]", r"Compiling \d+ files?"),
1013            ("[TX_HASH]", r"Transaction hash: 0x[0-9A-Fa-f]{64}"),
1014            ("[ADDRESS]", r"Address: 0x[0-9A-Fa-f]{40}"),
1015            ("[UPDATING_DEPENDENCIES]", r"Updating dependencies in .*"),
1016            ("[SAVED_TRANSACTIONS]", r"Transactions saved to: .*\.json"),
1017            ("[SAVED_SENSITIVE_VALUES]", r"Sensitive values saved to: .*\.json"),
1018            ("[ESTIMATED_GAS_PRICE]", r"Estimated gas price:\s*(\d+(\.\d+)?)\s*gwei"),
1019            ("[ESTIMATED_TOTAL_GAS_USED]", r"Estimated total gas used for script: \d+"),
1020            ("[ESTIMATED_AMOUNT_REQUIRED]", r"Estimated amount required:\s*(\d+(\.\d+)?)\s*ETH"),
1021        ];
1022        for (placeholder, re) in redactions {
1023            r.insert(placeholder, Regex::new(re).expect(re)).expect(re);
1024        }
1025        r
1026    });
1027    REDACTIONS.clone()
1028}
