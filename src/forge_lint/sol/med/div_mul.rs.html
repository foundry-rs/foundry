1use super::DivideBeforeMultiply;
2use crate::{
3    declare_forge_lint,
4    linter::{EarlyLintPass, LintContext},
5    sol::{Severity, SolLint},
6};
7use solar_ast::{BinOp, BinOpKind, Expr, ExprKind};
8
9declare_forge_lint!(
10    DIVIDE_BEFORE_MULTIPLY,
11    Severity::Med,
12    "divide-before-multiply",
13    "multiplication should occur before division to avoid loss of precision",
14    ""
15);
16
17impl<'ast> EarlyLintPass<'ast> for DivideBeforeMultiply {
18    fn check_expr(&mut self, ctx: &LintContext<'_>, expr: &'ast Expr<'ast>) {
19        if let ExprKind::Binary(left_expr, BinOp { kind: BinOpKind::Mul, .. }, _) = &expr.kind {
20            if contains_division(left_expr) {
21                ctx.emit(&DIVIDE_BEFORE_MULTIPLY, expr.span);
22            }
23        }
24    }
25}
26
27fn contains_division<'ast>(expr: &'ast Expr<'ast>) -> bool {
28    match &expr.kind {
29        ExprKind::Binary(_, BinOp { kind: BinOpKind::Div, .. }, _) => true,
30        ExprKind::Tuple(inner_exprs) => inner_exprs.iter().any(|opt_expr| {
31            if let Some(inner_expr) = opt_expr {
32                contains_division(inner_expr)
33            } else {
34                false
35            }
36        }),
37        _ => false,
38    }
39}
