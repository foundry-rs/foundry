19/// The [`revm::Inspector`] used when transacting in the evm
20#[derive(Clone, Debug, Default)]
21pub struct Inspector {
22    pub tracer: Option<TracingInspector>,
23    /// collects all `console.sol` logs
24    pub log_collector: Option<LogCollector>,
25}
26
27impl Inspector {
28    /// Called after the inspecting the evm
29    ///
30    /// This will log all `console.sol` logs
31    pub fn print_logs(&self) {
32        if let Some(collector) = &self.log_collector {
33            print_logs(&collector.logs);
34        }
35    }
36
37    /// Configures the `Tracer` [`revm::Inspector`]
38    pub fn with_tracing(mut self) -> Self {
39        self.tracer = Some(TracingInspector::new(TracingInspectorConfig::all().set_steps(false)));
40        self
41    }
42
43    pub fn with_tracing_config(mut self, config: TracingInspectorConfig) -> Self {
44        self.tracer = Some(TracingInspector::new(config));
45        self
46    }
47
48    /// Enables steps recording for `Tracer`.
49    pub fn with_steps_tracing(mut self) -> Self {
50        self.tracer = Some(TracingInspector::new(TracingInspectorConfig::all().with_state_diffs()));
51        self
52    }
53
54    /// Configures the `Tracer` [`revm::Inspector`]
55    pub fn with_log_collector(mut self) -> Self {
56        self.log_collector = Some(Default::default());
57        self
58    }
59}
60
61impl<DB: Database> revm::Inspector<DB> for Inspector {
62    fn initialize_interp(&mut self, interp: &mut Interpreter, ecx: &mut EvmContext<DB>) {
63        call_inspectors!([&mut self.tracer], |inspector| {
64            inspector.initialize_interp(interp, ecx);
65        });
66    }
67
68    fn step(&mut self, interp: &mut Interpreter, ecx: &mut EvmContext<DB>) {
69        call_inspectors!([&mut self.tracer], |inspector| {
70            inspector.step(interp, ecx);
71        });
72    }
73
74    fn step_end(&mut self, interp: &mut Interpreter, ecx: &mut EvmContext<DB>) {
75        call_inspectors!([&mut self.tracer], |inspector| {
76            inspector.step_end(interp, ecx);
77        });
78    }
79
80    fn log(&mut self, interp: &mut Interpreter, ecx: &mut EvmContext<DB>, log: &Log) {
81        call_inspectors!([&mut self.tracer, &mut self.log_collector], |inspector| {
82            inspector.log(interp, ecx, log);
83        });
84    }
85
86    fn call(&mut self, ecx: &mut EvmContext<DB>, inputs: &mut CallInputs) -> Option<CallOutcome> {
87        call_inspectors!(
88            #[ret]
89            [&mut self.tracer, &mut self.log_collector],
90            |inspector| inspector.call(ecx, inputs).map(Some),
91        );
92        None
93    }
94
95    fn call_end(
96        &mut self,
97        ecx: &mut EvmContext<DB>,
98        inputs: &CallInputs,
99        outcome: CallOutcome,
100    ) -> CallOutcome {
101        if let Some(tracer) = &mut self.tracer {
102            return tracer.call_end(ecx, inputs, outcome);
103        }
104
105        outcome
106    }
107
108    fn create(
109        &mut self,
110        ecx: &mut EvmContext<DB>,
111        inputs: &mut CreateInputs,
112    ) -> Option<CreateOutcome> {
113        if let Some(tracer) = &mut self.tracer {
114            if let Some(out) = tracer.create(ecx, inputs) {
115                return Some(out);
116            }
117        }
118        None
119    }
120
121    fn create_end(
122        &mut self,
123        ecx: &mut EvmContext<DB>,
124        inputs: &CreateInputs,
125        outcome: CreateOutcome,
126    ) -> CreateOutcome {
127        if let Some(tracer) = &mut self.tracer {
128            return tracer.create_end(ecx, inputs, outcome);
129        }
130
131        outcome
132    }
133
134    #[inline]
135    fn eofcreate(
136        &mut self,
137        ecx: &mut EvmContext<DB>,
138        inputs: &mut EOFCreateInputs,
139    ) -> Option<CreateOutcome> {
140        if let Some(tracer) = &mut self.tracer {
141            if let Some(out) = tracer.eofcreate(ecx, inputs) {
142                return Some(out);
143            }
144        }
145        None
146    }
147
148    #[inline]
149    fn eofcreate_end(
150        &mut self,
151        ecx: &mut EvmContext<DB>,
152        inputs: &EOFCreateInputs,
153        outcome: CreateOutcome,
154    ) -> CreateOutcome {
155        if let Some(tracer) = &mut self.tracer {
156            return tracer.eofcreate_end(ecx, inputs, outcome);
157        }
158
159        outcome
160    }
161
162    #[inline]
163    fn selfdestruct(&mut self, contract: Address, target: Address, value: U256) {
164        if let Some(tracer) = &mut self.tracer {
165            revm::Inspector::<DB>::selfdestruct(tracer, contract, target, value);
166        }
167    }
168}

170/// Prints all the logs
171pub fn print_logs(logs: &[Log]) {
172    for log in decode_console_logs(logs) {
173        tracing::info!(target: crate::logging::EVM_CONSOLE_LOG_TARGET, "{}", log);
174    }
