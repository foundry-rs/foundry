105/// Returns a [ProviderBuilder] instantiated using [Config] values.
106///
107/// Defaults to `http://localhost:8545` and `Mainnet`.
108pub fn get_provider_builder(config: &Config) -> Result<ProviderBuilder> {
109    let url = config.get_rpc_url_or_localhost_http()?;
110    let mut builder = ProviderBuilder::new(url.as_ref());
111
112    builder = builder.accept_invalid_certs(config.eth_rpc_accept_invalid_certs);
113
114    if let Ok(chain) = config.chain.unwrap_or_default().try_into() {
115        builder = builder.chain(chain);
116    }
117
118    if let Some(jwt) = config.get_rpc_jwt_secret()? {
119        builder = builder.jwt(jwt.as_ref());
120    }
121
122    if let Some(rpc_timeout) = config.eth_rpc_timeout {
123        builder = builder.timeout(Duration::from_secs(rpc_timeout));
124    }
125
126    if let Some(rpc_headers) = config.eth_rpc_headers.clone() {
127        builder = builder.headers(rpc_headers);
128    }
129
130    Ok(builder)
131}

145/// The amount can be tagged with a unit, e.g. "1ether".

150    Ok(if value.starts_with("0x") {

165/// Parses a `Duration` from a &str
166pub fn parse_delay(delay: &str) -> Result<Duration> {
167    let delay = if delay.ends_with("ms") {
168        let d: u64 = delay.trim_end_matches("ms").parse()?;
169        Duration::from_millis(d)
170    } else {
171        let d: f64 = delay.parse()?;
172        let delay = (d * 1000.0).round();
173        if delay.is_infinite() || delay.is_nan() || delay.is_sign_negative() {
174            eyre::bail!("delay must be finite and non-negative");
175        }
176
177        Duration::from_millis(delay as u64)
178    };
179    Ok(delay)
180}

187/// Loads a dotenv file, from the cwd and the project root, ignoring potential failure.
188///
189/// We could use `warn!` here, but that would imply that the dotenv file can't configure
190/// the logging behavior of Foundry.
191///
192/// Similarly, we could just use `eprintln!`, but colors are off limits otherwise dotenv is implied
193/// to not be able to configure the colors. It would also mess up the JSON output.
194pub fn load_dotenv() {
195    let load = |p: &Path| {
196        dotenvy::from_path(p.join(".env")).ok();
197    };
198
199    // we only want the .env file of the cwd and project root
200    // `find_project_root` calls `current_dir` internally so both paths are either both `Ok` or
201    // both `Err`
202    if let (Ok(cwd), Ok(prj_root)) = (std::env::current_dir(), find_project_root(None)) {
203        load(&prj_root);
204        if cwd != prj_root {
205            // prj root and cwd can be identical
206            load(&cwd);
207        }
208    };
209}

234/// Useful extensions to [`std::process::Command`].
235pub trait CommandUtils {
236    /// Returns the command's output if execution is successful, otherwise, throws an error.
237    fn exec(&mut self) -> Result<Output>;
238
239    /// Returns the command's stdout if execution is successful, otherwise, throws an error.
240    fn get_stdout_lossy(&mut self) -> Result<String>;
241}
242
243impl CommandUtils for Command {
244    #[track_caller]
245    fn exec(&mut self) -> Result<Output> {
246        trace!(command=?self, "executing");
247
248        let output = self.output()?;
249
250        trace!(code=?output.status.code(), ?output);
251
252        if output.status.success() {
253            Ok(output)
254        } else {
255            let stdout = String::from_utf8_lossy(&output.stdout);
256            let stdout = stdout.trim();
257            let stderr = String::from_utf8_lossy(&output.stderr);
258            let stderr = stderr.trim();
259            let msg = if stdout.is_empty() {
260                stderr.to_string()
261            } else if stderr.is_empty() {
262                stdout.to_string()
263            } else {
264                format!("stdout:\n{stdout}\n\nstderr:\n{stderr}")
265            };
266
267            let mut name = self.get_program().to_string_lossy();
268            if let Some(arg) = self.get_args().next() {
269                let arg = arg.to_string_lossy();
270                if !arg.starts_with('-') {
271                    let name = name.to_mut();
272                    name.push(' ');
273                    name.push_str(&arg);
274                }
275            }
276
277            let mut err = match output.status.code() {
278                Some(code) => format!("{name} exited with code {code}"),
279                None => format!("{name} terminated by a signal"),
280            };
281            if !msg.is_empty() {
282                err.push(':');
283                err.push(if msg.lines().count() == 0 { ' ' } else { '\n' });
284                err.push_str(&msg);
285            }
286            Err(eyre::eyre!(err))
287        }
288    }
289
290    #[track_caller]
291    fn get_stdout_lossy(&mut self) -> Result<String> {
292        let output = self.exec()?;
293        let stdout = String::from_utf8_lossy(&output.stdout);
294        Ok(stdout.trim().into())
295    }
296}

450    pub fn commit(self, msg: &str) -> Result<()> {
451        let output = self
452            .cmd()
453            .args(["commit", "-m", msg])
454            .args(cfg!(any(test, debug_assertions)).then_some("--no-gpg-sign"))
455            .output()?;
456        if !output.status.success() {

465                    stdout.trim(),
466                    stderr.trim()

468            }
469        }
470        Ok(())

552    /// Returns a list of tuples of submodule paths and their respective branches.
553    ///
554    /// This function reads the `.gitmodules` file and returns the paths of all submodules that have
555    /// a branch. The paths are relative to the Git::root_of(git.root) and not lib/ directory.
556    ///
557    /// `at` is the dir in which the `.gitmodules` file is located, this is the git root.
558    /// `lib` is name of the directory where the submodules are located.
559    pub fn read_submodules_with_branch(
560        self,
561        at: &Path,
562        lib: &OsStr,
563    ) -> Result<HashMap<PathBuf, String>> {
564        // Read the .gitmodules file
565        let gitmodules = foundry_common::fs::read_to_string(at.join(".gitmodules"))?;
566
567        let paths = SUBMODULE_BRANCH_REGEX
568            .captures_iter(&gitmodules)
569            .map(|cap| {
570                let path_str = cap.get(1).unwrap().as_str();
571                let path = PathBuf::from_str(path_str).unwrap();
572                trace!(path = %path.display(), "unstripped path");
573
574                // Keep only the components that come after the lib directory.
575                // This needs to be done because the lockfile uses paths relative foundry project
576                // root whereas .gitmodules use paths relative to the git root which may not be the
577                // project root. e.g monorepo.
578                // Hence, if path is lib/solady, then `lib/solady` is kept. if path is
579                // packages/contract-bedrock/lib/solady, then `lib/solady` is kept.
580                let lib_pos = path.components().find_position(|c| c.as_os_str() == lib);
581                let path = path
582                    .components()
583                    .skip(lib_pos.map(|(i, _)| i).unwrap_or(0))
584                    .collect::<PathBuf>();
585
586                let branch = cap.get(2).unwrap().as_str().to_string();
587                (path, branch)
588            })
589            .collect::<HashMap<_, _>>();
590
591        Ok(paths)
592    }

645        I: IntoIterator<Item = S>,
646        S: AsRef<OsStr>,
647    {
648        self.cmd()
649            .stderr(self.stderr())
650            .args(["submodule", "update", "--progress", "--init"])
651            .args(self.shallow.then_some("--depth=1"))
652            .args(force.then_some("--force"))
653            .args(remote.then_some("--remote"))
654            .args(no_fetch.then_some("--no-fetch"))
655            .args(recursive.then_some("--recursive"))
656            .args(paths)
657            .exec()
658            .map(drop)
659    }
660
661    pub fn submodule_foreach(self, recursive: bool, cmd: impl AsRef<OsStr>) -> Result<()> {
662        self.cmd()
663            .stderr(self.stderr())
664            .args(["submodule", "foreach"])
665            .args(recursive.then_some("--recursive"))
666            .arg(cmd)
667            .exec()
668            .map(drop)
669    }

671    /// If the status is prefix with `-`, the submodule is not initialized.
672    ///
673    /// Ref: <https://git-scm.com/docs/git-submodule#Documentation/git-submodule.txt-status--cached--recursive--ltpathgt82308203>
674    pub fn submodules_uninitialized(self) -> Result<bool> {
675        self.cmd()
676            .args(["submodule", "status"])
677            .get_stdout_lossy()
678            .map(|stdout| stdout.lines().any(|line| line.starts_with('-')))
679    }

681    /// Initializes the git submodules.
682    pub fn submodule_init(self) -> Result<()> {
683        self.cmd().stderr(self.stderr()).args(["submodule", "init"]).exec().map(drop)
684    }
685
686    pub fn submodules(&self) -> Result<Submodules> {
687        self.cmd().args(["submodule", "status"]).get_stdout_lossy().map(|stdout| stdout.parse())?
688    }
689
690    pub fn submodule_sync(self) -> Result<()> {
691        self.cmd().stderr(self.stderr()).args(["submodule", "sync"]).exec().map(drop)
692    }
693
694    pub fn cmd(self) -> Command {
695        let mut cmd = Self::cmd_no_root();
696        cmd.current_dir(self.root);
697        cmd
698    }
699
700    pub fn cmd_at(self, path: &Path) -> Command {
701        let mut cmd = Self::cmd_no_root();
702        cmd.current_dir(path);
703        cmd
704    }
705
706    pub fn cmd_no_root() -> Command {
707        let mut cmd = Command::new("git");
708        cmd.stdout(Stdio::piped()).stderr(Stdio::piped());
709        cmd
710    }

712    // don't set this in cmd() because it's not wanted for all commands
713    fn stderr(self) -> Stdio {
714        if self.quiet { Stdio::piped() } else { Stdio::inherit() }
715    }

718/// Deserialized `git submodule status lib/dep` output.
719#[derive(Debug, Clone, serde::Serialize, serde::Deserialize, PartialEq, Eq)]
720pub struct Submodule {
721    /// Current commit hash the submodule is checked out at.
722    rev: String,
723    /// Relative path to the submodule.
724    path: PathBuf,
725}
726
727impl Submodule {
728    pub fn new(rev: String, path: PathBuf) -> Self {
729        Self { rev, path }
730    }
731
732    pub fn rev(&self) -> &str {
733        &self.rev
734    }
735
736    pub fn path(&self) -> &PathBuf {
737        &self.path
738    }
739}
740
741impl FromStr for Submodule {
742    type Err = eyre::Report;
743
744    fn from_str(s: &str) -> Result<Self> {
745        let caps = SUBMODULE_STATUS_REGEX
746            .captures(s)
747            .ok_or_else(|| eyre::eyre!("Invalid submodule status format"))?;
748
749        Ok(Self {
750            rev: caps.get(1).unwrap().as_str().to_string(),
751            path: PathBuf::from(caps.get(2).unwrap().as_str()),
752        })
753    }
754}
755
756/// Deserialized `git submodule status` output.
757#[derive(Debug, Clone, PartialEq, Eq)]
758pub struct Submodules(pub Vec<Submodule>);
759
760impl Submodules {
761    pub fn len(&self) -> usize {
762        self.0.len()
763    }
764
765    pub fn is_empty(&self) -> bool {
766        self.0.is_empty()
767    }
768}
769
770impl FromStr for Submodules {
771    type Err = eyre::Report;
772
773    fn from_str(s: &str) -> Result<Self> {
774        let subs = s.lines().map(str::parse).collect::<Result<Vec<Submodule>>>()?;
775        Ok(Self(subs))
776    }
777}

779impl<'a> IntoIterator for &'a Submodules {
780    type Item = &'a Submodule;
781    type IntoIter = std::slice::Iter<'a, Submodule>;
782
783    fn into_iter(self) -> Self::IntoIter {
784        self.0.iter()
785    }
