167    fn apply(&self, state: &mut Cheatcodes) -> Result {
168        let Self { json, path } = self;
169        let value =
170            serde_json::from_str(json).unwrap_or_else(|_| JsonValue::String(json.to_owned()));

172        let toml_string = format_json_to_toml(value)?;
173        super::fs::write_file(state, path.as_ref(), toml_string.as_bytes())

177impl Cheatcode for writeToml_1Call {
178    fn apply(&self, state: &mut Cheatcodes) -> Result {
179        let Self { json, path, valueKey } = self;
180        let json =
181            serde_json::from_str(json).unwrap_or_else(|_| JsonValue::String(json.to_owned()));
182
183        let data_path = state.config.ensure_path_allowed(path, FsAccessKind::Read)?;
184        let toml_data = fs::read_to_string(data_path)?;
185        let json_data: JsonValue =
186            toml::from_str(&toml_data).map_err(|e| fmt_err!("failed parsing TOML: {e}"))?;
187        let value =
188            jsonpath_lib::replace_with(json_data, &canonicalize_json_path(valueKey), &mut |_| {
189                Some(json.clone())
190            })?;
191
192        let toml_string = format_json_to_toml(value)?;
193        super::fs::write_file(state, path.as_ref(), toml_string.as_bytes())
194    }
195}

197impl Cheatcode for writeTomlUpsertCall {
198    fn apply(&self, state: &mut Cheatcodes) -> Result {
199        let Self { json: value, path, valueKey } = self;
200
201        // Read and parse the TOML file
202        let data_path = state.config.ensure_path_allowed(path, FsAccessKind::Read)?;
203        let toml_data = fs::read_to_string(&data_path)?;
204
205        // Convert to JSON and update the object
206        let mut json_data: JsonValue =
207            toml::from_str(&toml_data).map_err(|e| fmt_err!("failed parsing TOML: {e}"))?;
208        upsert_json_value(&mut json_data, value, valueKey)?;
209
210        // Serialize back to TOML and write the updated content back to the file
211        let toml_string = format_json_to_toml(json_data)?;
212        super::fs::write_file(state, path.as_ref(), toml_string.as_bytes())
213    }
214}

216/// Parse
217fn parse_toml_str(toml: &str) -> Result<TomlValue> {
218    toml::from_str(toml).map_err(|e| fmt_err!("failed parsing TOML: {e}"))
219}
220
221/// Parse a TOML string and return the value at the given path.
222fn parse_toml(toml: &str, key: &str) -> Result {
223    parse_json(&toml_to_json_string(toml)?, key)
224}
225
226/// Parse a TOML string and return the value at the given path, coercing it to the given type.
227fn parse_toml_coerce(toml: &str, key: &str, ty: &DynSolType) -> Result {
228    parse_json_coerce(&toml_to_json_string(toml)?, key, ty)
229}
230
231/// Parse a TOML string and return an array of all keys at the given path.
232fn parse_toml_keys(toml: &str, key: &str) -> Result {
233    parse_json_keys(&toml_to_json_string(toml)?, key)
234}
235
236/// Convert a TOML string to a JSON string.
237fn toml_to_json_string(toml: &str) -> Result<String> {
238    let toml = parse_toml_str(toml)?;
239    let json = toml_to_json_value(toml);
240    serde_json::to_string(&json).map_err(|e| fmt_err!("failed to serialize JSON: {e}"))
241}
242
243/// Format a JSON value to a TOML pretty string.
244fn format_json_to_toml(json: JsonValue) -> Result<String> {
245    let toml = json_to_toml_value(json);
246    toml::to_string_pretty(&toml).map_err(|e| fmt_err!("failed to serialize TOML: {e}"))
247}
248
249/// Convert a TOML value to a JSON value.
250pub(super) fn toml_to_json_value(toml: TomlValue) -> JsonValue {
251    match toml {
252        TomlValue::String(s) => match s.as_str() {
253            "null" => JsonValue::Null,
254            _ => JsonValue::String(s),
255        },
256        TomlValue::Integer(i) => JsonValue::Number(i.into()),
257        TomlValue::Float(f) => JsonValue::Number(serde_json::Number::from_f64(f).unwrap()),
258        TomlValue::Boolean(b) => JsonValue::Bool(b),
259        TomlValue::Array(a) => JsonValue::Array(a.into_iter().map(toml_to_json_value).collect()),
260        TomlValue::Table(t) => {
261            JsonValue::Object(t.into_iter().map(|(k, v)| (k, toml_to_json_value(v))).collect())
262        }
263        TomlValue::Datetime(d) => JsonValue::String(d.to_string()),
264    }
265}
266
267/// Convert a JSON value to a TOML value.
268fn json_to_toml_value(json: JsonValue) -> TomlValue {
269    match json {
270        JsonValue::String(s) => TomlValue::String(s),
271        JsonValue::Number(n) => match n.as_i64() {
272            Some(i) => TomlValue::Integer(i),
273            None => match n.as_f64() {
274                Some(f) => TomlValue::Float(f),
275                None => TomlValue::String(n.to_string()),
276            },
277        },
278        JsonValue::Bool(b) => TomlValue::Boolean(b),
279        JsonValue::Array(a) => TomlValue::Array(a.into_iter().map(json_to_toml_value).collect()),
280        JsonValue::Object(o) => {
281            TomlValue::Table(o.into_iter().map(|(k, v)| (k, json_to_toml_value(v))).collect())
282        }
283        JsonValue::Null => TomlValue::String("null".to_string()),
284    }
