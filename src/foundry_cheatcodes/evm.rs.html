36/// Records storage slots reads and writes.
37#[derive(Clone, Debug, Default)]
38pub struct RecordAccess {
39    /// Storage slots reads.
40    pub reads: HashMap<Address, Vec<U256>>,
41    /// Storage slots writes.
42    pub writes: HashMap<Address, Vec<U256>>,
43}
44
45impl RecordAccess {
46    /// Records a read access to a storage slot.
47    pub fn record_read(&mut self, target: Address, slot: U256) {
48        self.reads.entry(target).or_default().push(slot);
49    }
50
51    /// Records a write access to a storage slot.
52    ///
53    /// This also records a read internally as `SSTORE` does an implicit `SLOAD`.
54    pub fn record_write(&mut self, target: Address, slot: U256) {
55        self.record_read(target, slot);
56        self.writes.entry(target).or_default().push(slot);
57    }
58}

104/// Account state diff info.
105#[derive(Serialize, Default)]
106#[serde(rename_all = "camelCase")]
107struct AccountStateDiffs {
108    /// Address label, if any set.
109    label: Option<String>,
110    /// Account balance changes.
111    balance_diff: Option<BalanceDiff>,
112    /// State changes, per slot.
113    state_diff: BTreeMap<B256, SlotStateDiff>,
114}
115
116impl Display for AccountStateDiffs {
117    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> eyre::Result<(), std::fmt::Error> {
118        // Print changed account.
119        if let Some(label) = &self.label {
120            writeln!(f, "label: {label}")?;
121        }
122        // Print balance diff if changed.
123        if let Some(balance_diff) = &self.balance_diff {
124            if balance_diff.previous_value != balance_diff.new_value {
125                writeln!(
126                    f,
127                    "- balance diff: {} → {}",
128                    balance_diff.previous_value, balance_diff.new_value
129                )?;
130            }
131        }
132        // Print state diff if any.
133        if !&self.state_diff.is_empty() {
134            writeln!(f, "- state diff:")?;
135            for (slot, slot_changes) in &self.state_diff {
136                writeln!(
137                    f,
138                    "@ {slot}: {} → {}",
139                    slot_changes.previous_value, slot_changes.new_value
140                )?;
141            }
142        }
143
144        Ok(())
145    }
146}

170impl Cheatcode for loadCall {
171    fn apply_stateful(&self, ccx: &mut CheatsCtxt) -> Result {
172        let Self { target, slot } = *self;
173        ensure_not_precompile!(&target, ccx);
174        ccx.ecx.load_account(target)?;
175        let mut val = ccx.ecx.sload(target, slot.into())?;
176
177        if val.is_cold && val.data.is_zero() {
178            if ccx.state.has_arbitrary_storage(&target) {
179                // If storage slot is untouched and load from a target with arbitrary storage,
180                // then set random value for current slot.
181                let rand_value = ccx.state.rng().gen();
182                ccx.state.arbitrary_storage.as_mut().unwrap().save(
183                    ccx.ecx,
184                    target,
185                    slot.into(),
186                    rand_value,
187                );
188                val.data = rand_value;
189            } else if ccx.state.is_arbitrary_storage_copy(&target) {
190                // If storage slot is untouched and load from a target that copies storage from
191                // a source address with arbitrary storage, then copy existing arbitrary value.
192                // If no arbitrary value generated yet, then the random one is saved and set.
193                let rand_value = ccx.state.rng().gen();
194                val.data = ccx.state.arbitrary_storage.as_mut().unwrap().copy(
195                    ccx.ecx,
196                    target,
197                    slot.into(),
198                    rand_value,
199                );
200            }
201        }
202
203        Ok(val.abi_encode())
204    }
205}

207impl Cheatcode for loadAllocsCall {
208    fn apply_stateful(&self, ccx: &mut CheatsCtxt) -> Result {
209        let Self { pathToAllocsJson } = self;
210
211        let path = Path::new(pathToAllocsJson);
212        ensure!(path.exists(), "allocs file does not exist: {pathToAllocsJson}");
213
214        // Let's first assume we're reading a file with only the allocs.
215        let allocs: BTreeMap<Address, GenesisAccount> = match read_json_file(path) {
216            Ok(allocs) => allocs,
217            Err(_) => {
218                // Let's try and read from a genesis file, and extract allocs.
219                let genesis = read_json_file::<Genesis>(path)?;
220                genesis.alloc
221            }
222        };
223
224        // Then, load the allocs into the database.
225        ccx.ecx
226            .db
227            .load_allocs(&allocs, &mut ccx.ecx.journaled_state)
228            .map(|()| Vec::default())
229            .map_err(|e| fmt_err!("failed to load allocs: {e}"))
230    }
231}

249impl Cheatcode for dumpStateCall {
250    fn apply_stateful(&self, ccx: &mut CheatsCtxt) -> Result {
251        let Self { pathToStateJson } = self;
252        let path = Path::new(pathToStateJson);
253
254        // Do not include system account or empty accounts in the dump.
255        let skip = |key: &Address, val: &Account| {
256            key == &CHEATCODE_ADDRESS ||
257                key == &CALLER ||
258                key == &HARDHAT_CONSOLE_ADDRESS ||
259                key == &TEST_CONTRACT_ADDRESS ||
260                key == &ccx.caller ||
261                key == &ccx.state.config.evm_opts.sender ||
262                val.is_empty()
263        };
264
265        let alloc = ccx
266            .ecx
267            .journaled_state
268            .state()
269            .iter_mut()
270            .filter(|(key, val)| !skip(key, val))
271            .map(|(key, val)| (key, genesis_account(val)))
272            .collect::<BTreeMap<_, _>>();
273
274        write_json_file(path, &alloc)?;
275        Ok(Default::default())
276    }
277}

525impl Cheatcode for resetNonceCall {
526    fn apply_stateful(&self, ccx: &mut CheatsCtxt) -> Result {
527        let Self { account } = self;
528        let account = journaled_account(ccx.ecx, *account)?;
529        // Per EIP-161, EOA nonces start at 0, but contract nonces
530        // start at 1. Comparing by code_hash instead of code
531        // to avoid hitting the case where account's code is None.
532        let empty = account.info.code_hash == KECCAK_EMPTY;
533        let nonce = if empty { 0 } else { 1 };
534        account.info.nonce = nonce;
535        debug!(target: "cheatcodes", nonce, "reset");
536        Ok(Default::default())
537    }
538}

814impl Cheatcode for startDebugTraceRecordingCall {
815    fn apply_full(&self, ccx: &mut CheatsCtxt, executor: &mut dyn CheatcodesExecutor) -> Result {
816        let Some(tracer) = executor.tracing_inspector().and_then(|t| t.as_mut()) else {
817            return Err(Error::from("no tracer initiated, consider adding -vvv flag"))
818        };
819
820        let mut info = RecordDebugStepInfo {
821            // will be updated later
822            start_node_idx: 0,
823            // keep the original config to revert back later
824            original_tracer_config: *tracer.config(),
825        };
826
827        // turn on tracer configuration for recording
828        tracer.update_config(|config| {
829            config
830                .set_steps(true)
831                .set_memory_snapshots(true)
832                .set_stack_snapshots(StackSnapshotType::Full)
833        });
834
835        // track where the recording starts
836        if let Some(last_node) = tracer.traces().nodes().last() {
837            info.start_node_idx = last_node.idx;
838        }
839
840        ccx.state.record_debug_steps_info = Some(info);
841        Ok(Default::default())
842    }

845impl Cheatcode for stopAndReturnDebugTraceRecordingCall {
846    fn apply_full(&self, ccx: &mut CheatsCtxt, executor: &mut dyn CheatcodesExecutor) -> Result {
847        let Some(tracer) = executor.tracing_inspector().and_then(|t| t.as_mut()) else {
848            return Err(Error::from("no tracer initiated, consider adding -vvv flag"))
849        };
850
851        let Some(record_info) = ccx.state.record_debug_steps_info else {
852            return Err(Error::from("nothing recorded"))
853        };
854
855        // Use the trace nodes to flatten the call trace
856        let root = tracer.traces();
857        let steps = flatten_call_trace(0, root, record_info.start_node_idx);
858
859        let debug_steps: Vec<DebugStep> =
860            steps.iter().map(|&step| convert_call_trace_to_debug_step(step)).collect();
861        // Free up memory by clearing the steps if they are not recorded outside of cheatcode usage.
862        if !record_info.original_tracer_config.record_steps {
863            tracer.traces_mut().nodes_mut().iter_mut().for_each(|node| {
864                node.trace.steps = Vec::new();
865                node.logs = Vec::new();
866                node.ordering = Vec::new();
867            });
868        }
869
870        // Revert the tracer config to the one before recording
871        tracer.update_config(|_config| record_info.original_tracer_config);
872
873        // Clean up the recording info
874        ccx.state.record_debug_steps_info = None;
875
876        Ok(debug_steps.abi_encode())
877    }

957fn inner_start_gas_snapshot(
958    ccx: &mut CheatsCtxt,
959    group: Option<String>,
960    name: Option<String>,
961) -> Result {
962    // Revert if there is an active gas snapshot as we can only have one active snapshot at a time.
963    if ccx.state.gas_metering.active_gas_snapshot.is_some() {
964        let (group, name) = ccx.state.gas_metering.active_gas_snapshot.as_ref().unwrap().clone();
965        bail!("gas snapshot was already started with group: {group} and name: {name}");
966    }
967
968    let (group, name) = derive_snapshot_name(ccx, group, name);
969
970    ccx.state.gas_metering.gas_records.push(GasRecord {
971        group: group.clone(),
972        name: name.clone(),
973        gas_used: 0,
974        depth: ccx.ecx.journaled_state.depth(),
975    });
976
977    ccx.state.gas_metering.active_gas_snapshot = Some((group, name));
978
979    ccx.state.gas_metering.start();
980
981    Ok(Default::default())
982}

984fn inner_stop_gas_snapshot(
985    ccx: &mut CheatsCtxt,
986    group: Option<String>,
987    name: Option<String>,
988) -> Result {
989    // If group and name are not provided, use the last snapshot group and name.
990    let (group, name) = group.zip(name).unwrap_or_else(|| {
991        let (group, name) = ccx.state.gas_metering.active_gas_snapshot.as_ref().unwrap().clone();
992        (group, name)
993    });
994
995    if let Some(record) = ccx
996        .state
997        .gas_metering
998        .gas_records
999        .iter_mut()
1000        .find(|record| record.group == group && record.name == name)
1001    {
1002        // Calculate the gas used since the snapshot was started.
1003        // We subtract 171 from the gas used to account for gas used by the snapshot itself.
1004        let value = record.gas_used.saturating_sub(171);
1005
1006        ccx.state
1007            .gas_snapshots
1008            .entry(group.clone())
1009            .or_default()
1010            .insert(name.clone(), value.to_string());
1011
1012        // Stop the gas metering.
1013        ccx.state.gas_metering.stop();
1014
1015        // Remove the gas record.
1016        ccx.state
1017            .gas_metering
1018            .gas_records
1019            .retain(|record| record.group != group && record.name != name);
1020
1021        // Clear last snapshot cache if we have an exact match.
1022        if let Some((snapshot_group, snapshot_name)) = &ccx.state.gas_metering.active_gas_snapshot {
1023            if snapshot_group == &group && snapshot_name == &name {
1024                ccx.state.gas_metering.active_gas_snapshot = None;
1025            }
1026        }
1027
1028        Ok(value.abi_encode())
1029    } else {
1030        bail!("no gas snapshot was started with the name: {name} in group: {group}");
1031    }
1032}

1047/// Reads the current caller information and returns the current [CallerMode], `msg.sender` and
1048/// `tx.origin`.
1049///
1050/// Depending on the current caller mode, one of the following results will be returned:
1051/// - If there is an active prank:
1052///     - caller_mode will be equal to:
1053///         - [CallerMode::Prank] if the prank has been set with `vm.prank(..)`.
1054///         - [CallerMode::RecurrentPrank] if the prank has been set with `vm.startPrank(..)`.
1055///     - `msg.sender` will be equal to the address set for the prank.
1056///     - `tx.origin` will be equal to the default sender address unless an alternative one has been
1057///       set when configuring the prank.
1058///
1059/// - If there is an active broadcast:
1060///     - caller_mode will be equal to:
1061///         - [CallerMode::Broadcast] if the broadcast has been set with `vm.broadcast(..)`.
1062///         - [CallerMode::RecurrentBroadcast] if the broadcast has been set with
1063///           `vm.startBroadcast(..)`.
1064///     - `msg.sender` and `tx.origin` will be equal to the address provided when setting the
1065///       broadcast.
1066///
1067/// - If no caller modification is active:
1068///     - caller_mode will be equal to [CallerMode::None],
1069///     - `msg.sender` and `tx.origin` will be equal to the default sender address.
1070fn read_callers(state: &Cheatcodes, default_sender: &Address) -> Result {
1071    let Cheatcodes { prank, broadcast, .. } = state;
1072
1073    let mut mode = CallerMode::None;
1074    let mut new_caller = default_sender;
1075    let mut new_origin = default_sender;
1076    if let Some(prank) = prank {
1077        mode = if prank.single_call { CallerMode::Prank } else { CallerMode::RecurrentPrank };
1078        new_caller = &prank.new_caller;
1079        if let Some(new) = &prank.new_origin {
1080            new_origin = new;
1081        }
1082    } else if let Some(broadcast) = broadcast {
1083        mode = if broadcast.single_call {
1084            CallerMode::Broadcast
1085        } else {
1086            CallerMode::RecurrentBroadcast
1087        };
1088        new_caller = &broadcast.new_origin;
1089        new_origin = &broadcast.new_origin;
1090    }
1091
1092    Ok((mode, new_caller, new_origin).abi_encode_params())
1093}

1140/// Helper function to returns state diffs recorded for each changed account.
1141fn get_recorded_state_diffs(state: &mut Cheatcodes) -> BTreeMap<Address, AccountStateDiffs> {
1142    let mut state_diffs: BTreeMap<Address, AccountStateDiffs> = BTreeMap::default();
1143    if let Some(records) = &state.recorded_account_diffs_stack {
1144        records
1145            .iter()
1146            .flatten()
1147            .filter(|account_access| {
1148                !account_access.storageAccesses.is_empty() ||
1149                    account_access.oldBalance != account_access.newBalance
1150            })
1151            .for_each(|account_access| {
1152                // Record account balance diffs.
1153                if account_access.oldBalance != account_access.newBalance {
1154                    let account_diff =
1155                        state_diffs.entry(account_access.account).or_insert_with(|| {
1156                            AccountStateDiffs {
1157                                label: state.labels.get(&account_access.account).cloned(),
1158                                ..Default::default()
1159                            }
1160                        });
1161                    // Update balance diff. Do not overwrite the initial balance if already set.
1162                    if let Some(diff) = &mut account_diff.balance_diff {
1163                        diff.new_value = account_access.newBalance;
1164                    } else {
1165                        account_diff.balance_diff = Some(BalanceDiff {
1166                            previous_value: account_access.oldBalance,
1167                            new_value: account_access.newBalance,
1168                        });
1169                    }
1170                }
1171
1172                // Record account state diffs.
1173                for storage_access in &account_access.storageAccesses {
1174                    if storage_access.isWrite && !storage_access.reverted {
1175                        let account_diff = state_diffs
1176                            .entry(storage_access.account)
1177                            .or_insert_with(|| AccountStateDiffs {
1178                                label: state.labels.get(&storage_access.account).cloned(),
1179                                ..Default::default()
1180                            });
1181                        // Update state diff. Do not overwrite the initial value if already set.
1182                        match account_diff.state_diff.entry(storage_access.slot) {
1183                            Entry::Vacant(slot_state_diff) => {
1184                                slot_state_diff.insert(SlotStateDiff {
1185                                    previous_value: storage_access.previousValue,
1186                                    new_value: storage_access.newValue,
1187                                });
1188                            }
1189                            Entry::Occupied(mut slot_state_diff) => {
1190                                slot_state_diff.get_mut().new_value = storage_access.newValue;
1191                            }
1192                        }
1193                    }
1194                }
1195            });
1196    }
1197    state_diffs
1198}
