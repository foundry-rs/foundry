92/// Helper function to attach an EIP-7702 delegation.
93fn attach_delegation(
94    ccx: &mut CheatsCtxt,
95    delegation: &SignedDelegation,
96    cross_chain: bool,
97) -> Result {
98    let SignedDelegation { v, r, s, nonce, implementation } = delegation;
99    // Set chain id to 0 if universal deployment is preferred.
100    // See https://github.com/ethereum/EIPs/blob/master/EIPS/eip-7702.md#protection-from-malleability-cross-chain
101    let chain_id = if cross_chain { U256::from(0) } else { U256::from(ccx.ecx.env.cfg.chain_id) };
102
103    let auth = Authorization { address: *implementation, nonce: *nonce, chain_id };
104    let signed_auth = SignedAuthorization::new_unchecked(
105        auth,
106        *v,
107        U256::from_be_bytes(r.0),
108        U256::from_be_bytes(s.0),
109    );
110    write_delegation(ccx, signed_auth.clone())?;
111    ccx.state.active_delegation = Some(signed_auth);
112    Ok(Default::default())
113}

115/// Helper function to sign and attach (if needed) an EIP-7702 delegation.
116/// Uses the provided nonce, otherwise retrieves and increments the nonce of the EOA.
117fn sign_delegation(
118    ccx: &mut CheatsCtxt,
119    private_key: Uint<256, 4>,
120    implementation: Address,
121    nonce: Option<u64>,
122    cross_chain: bool,
123    attach: bool,
124) -> Result<Vec<u8>> {
125    let signer = PrivateKeySigner::from_bytes(&B256::from(private_key))?;
126    let nonce = if let Some(nonce) = nonce {
127        nonce
128    } else {
129        let authority_acc =
130            ccx.ecx.journaled_state.load_account(signer.address(), &mut ccx.ecx.db)?;
131        // If we don't have a nonce then use next auth account nonce.
132        authority_acc.data.info.nonce + 1
133    };
134    let chain_id = if cross_chain { U256::from(0) } else { U256::from(ccx.ecx.env.cfg.chain_id) };
135
136    let auth = Authorization { address: implementation, nonce, chain_id };
137    let sig = signer.sign_hash_sync(&auth.signature_hash())?;
138    // Attach delegation.
139    if attach {
140        let signed_auth = SignedAuthorization::new_unchecked(auth, sig.v() as u8, sig.r(), sig.s());
141        write_delegation(ccx, signed_auth.clone())?;
142        ccx.state.active_delegation = Some(signed_auth);
143    }
144    Ok(SignedDelegation {
145        v: sig.v() as u8,
146        r: sig.r().into(),
147        s: sig.s().into(),
148        nonce,
149        implementation,
150    }
151    .abi_encode())
152}
153
154fn write_delegation(ccx: &mut CheatsCtxt, auth: SignedAuthorization) -> Result<()> {
155    let authority = auth.recover_authority().map_err(|e| format!("{e}"))?;
156    let authority_acc = ccx.ecx.journaled_state.load_account(authority, &mut ccx.ecx.db)?;
157
158    if authority_acc.data.info.nonce + 1 != auth.nonce {
159        return Err("invalid nonce".into());
160    }
161
162    if auth.address.is_zero() {
163        // Set empty code if the delegation address of authority is 0x.
164        // See https://github.com/ethereum/EIPs/blob/master/EIPS/eip-7702.md#behavior.
165        ccx.ecx.journaled_state.set_code_with_hash(authority, Bytecode::default(), KECCAK_EMPTY);
166    } else {
167        let bytecode = Bytecode::new_eip7702(*auth.address());
168        ccx.ecx.journaled_state.set_code(authority, bytecode);
169    }
170    Ok(())
171}

173impl Cheatcode for attachBlobCall {
174    fn apply_stateful(&self, ccx: &mut CheatsCtxt) -> Result {
175        let Self { blob } = self;
176        ensure!(
177            ccx.ecx.spec_id() >= SpecId::CANCUN,
178            "`attachBlob` is not supported before the Cancun hard fork; \
179             see EIP-4844: https://eips.ethereum.org/EIPS/eip-4844"
180        );
181        let sidecar: SidecarBuilder<SimpleCoder> = SidecarBuilder::from_slice(blob);
182        let sidecar = sidecar.build().map_err(|e| format!("{e}"))?;
183        ccx.state.active_blob_sidecar = Some(sidecar);
184        Ok(Default::default())
185    }
186}

228pub struct Broadcast {

233    /// Original `tx.origin`

241/// Contains context for wallet management.
242#[derive(Debug)]
243pub struct WalletsInner {
244    /// All signers in scope of the script.
245    pub multi_wallet: MultiWallet,
246    /// Optional signer provided as `--sender` flag.
247    pub provided_sender: Option<Address>,
248}
249
250/// Clonable wrapper around [`WalletsInner`].
251#[derive(Debug, Clone)]
252pub struct Wallets {
253    /// Inner data.
254    pub inner: Arc<Mutex<WalletsInner>>,
255}
256
257impl Wallets {
258    #[expect(missing_docs)]
259    pub fn new(multi_wallet: MultiWallet, provided_sender: Option<Address>) -> Self {
260        Self { inner: Arc::new(Mutex::new(WalletsInner { multi_wallet, provided_sender })) }
261    }
262
263    /// Consumes [Wallets] and returns [MultiWallet].
264    ///
265    /// Panics if [Wallets] is still in use.
266    pub fn into_multi_wallet(self) -> MultiWallet {
267        Arc::into_inner(self.inner)
268            .map(|m| m.into_inner().multi_wallet)
269            .unwrap_or_else(|| panic!("not all instances were dropped"))
270    }
271
272    /// Locks inner Mutex and adds a signer to the [MultiWallet].
273    pub fn add_private_key(&self, private_key: &B256) -> Result<()> {
274        self.add_local_signer(PrivateKeySigner::from_bytes(private_key)?);
275        Ok(())
276    }
277
278    /// Locks inner Mutex and adds a signer to the [MultiWallet].
279    pub fn add_local_signer(&self, wallet: PrivateKeySigner) {
280        self.inner.lock().multi_wallet.add_signer(WalletSigner::Local(wallet));
281    }
282
283    /// Locks inner Mutex and returns all signer addresses in the [MultiWallet].
284    pub fn signers(&self) -> Result<Vec<Address>> {
285        Ok(self.inner.lock().multi_wallet.signers()?.keys().cloned().collect())
286    }
287
288    /// Number of signers in the [MultiWallet].
289    pub fn len(&self) -> usize {
290        let mut inner = self.inner.lock();
291        let signers = inner.multi_wallet.signers();
292        if signers.is_err() {
293            return 0;
294        }
295        signers.unwrap().len()
296    }
297
298    /// Whether the [MultiWallet] is empty.
299    pub fn is_empty(&self) -> bool {
300        self.len() == 0
301    }
302}

304/// Sets up broadcasting from a script using `new_origin` as the sender.
305fn broadcast(ccx: &mut CheatsCtxt, new_origin: Option<&Address>, single_call: bool) -> Result {
306    let depth = ccx.ecx.journaled_state.depth();
307    ensure!(
308        ccx.state.get_prank(depth).is_none(),
309        "you have an active prank; broadcasting and pranks are not compatible"
310    );
311    ensure!(ccx.state.broadcast.is_none(), "a broadcast is active already");
312
313    let mut new_origin = new_origin.cloned();
314
315    if new_origin.is_none() {
316        let mut wallets = ccx.state.wallets().inner.lock();
317        if let Some(provided_sender) = wallets.provided_sender {
318            new_origin = Some(provided_sender);
319        } else {
320            let signers = wallets.multi_wallet.signers()?;
321            if signers.len() == 1 {
322                let address = signers.keys().next().unwrap();
323                new_origin = Some(*address);
324            }
325        }
326    }
327
328    let broadcast = Broadcast {
329        new_origin: new_origin.unwrap_or(ccx.ecx.env.tx.caller),
330        original_caller: ccx.caller,
331        original_origin: ccx.ecx.env.tx.caller,
332        depth,
333        single_call,
334    };
335    debug!(target: "cheatcodes", ?broadcast, "started");
336    ccx.state.broadcast = Some(broadcast);
337    Ok(Default::default())
338}

340/// Sets up broadcasting from a script with the sender derived from `private_key`.
341/// Adds this private key to `state`'s `wallets` vector to later be used for signing
342/// if broadcast is successful.
343fn broadcast_key(ccx: &mut CheatsCtxt, private_key: &U256, single_call: bool) -> Result {
344    let wallet = super::crypto::parse_wallet(private_key)?;
345    let new_origin = wallet.address();
346
347    let result = broadcast(ccx, Some(&new_origin), single_call);
348    if result.is_ok() {
349        let wallets = ccx.state.wallets();
350        wallets.add_local_signer(wallet);
351    }
352    result
