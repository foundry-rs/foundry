4/// Prank information.
5#[derive(Clone, Debug, Default)]
6pub struct Prank {
7    /// Address of the contract that initiated the prank
8    pub prank_caller: Address,
9    /// Address of `tx.origin` when the prank was initiated
10    pub prank_origin: Address,
11    /// The address to assign to `msg.sender`
12    pub new_caller: Address,
13    /// The address to assign to `tx.origin`
14    pub new_origin: Option<Address>,
15    /// The depth at which the prank was called
16    pub depth: u64,
17    /// Whether the prank stops by itself after the next call
18    pub single_call: bool,
19    /// Whether the prank should be be applied to delegate call
20    pub delegate_call: bool,
21    /// Whether the prank has been used yet (false if unused)
22    pub used: bool,
23}
24
25impl Prank {
26    /// Create a new prank.
27    pub fn new(
28        prank_caller: Address,
29        prank_origin: Address,
30        new_caller: Address,
31        new_origin: Option<Address>,
32        depth: u64,
33        single_call: bool,
34        delegate_call: bool,
35    ) -> Self {
36        Self {
37            prank_caller,
38            prank_origin,
39            new_caller,
40            new_origin,
41            depth,
42            single_call,
43            delegate_call,
44            used: false,
45        }
46    }
47
48    /// Apply the prank by setting `used` to true iff it is false
49    /// Only returns self in the case it is updated (first application)
50    pub fn first_time_applied(&self) -> Option<Self> {
51        if self.used {
52            None
53        } else {
54            Some(Self { used: true, ..self.clone() })
55        }
56    }
57}

123fn prank(
124    ccx: &mut CheatsCtxt,
125    new_caller: &Address,
126    new_origin: Option<&Address>,
127    single_call: bool,
128    delegate_call: bool,
129) -> Result {
130    let prank = Prank::new(
131        ccx.caller,
132        ccx.ecx.env.tx.caller,
133        *new_caller,
134        new_origin.copied(),
135        ccx.ecx.journaled_state.depth(),
136        single_call,
137        delegate_call,
138    );
139
140    // Ensure that code exists at `msg.sender` if delegate calling.
141    if delegate_call {
142        let code = ccx.code(*new_caller)?;
143        ensure!(!code.is_empty(), "cannot `prank` delegate call from an EOA");
144    }
145
146    if let Some(Prank { used, single_call: current_single_call, .. }) = ccx.state.prank {
147        ensure!(used, "cannot overwrite a prank until it is applied at least once");
148        // This case can only fail if the user calls `vm.startPrank` and then `vm.prank` later on.
149        // This should not be possible without first calling `stopPrank`
150        ensure!(
151            single_call == current_single_call,
152            "cannot override an ongoing prank with a single vm.prank; \
153             use vm.startPrank to override the current prank"
154        );
155    }
156
157    ensure!(
158        ccx.state.broadcast.is_none(),
159        "cannot `prank` for a broadcasted transaction; \
160         pass the desired `tx.origin` into the `broadcast` cheatcode call"
161    );
162
163    ccx.state.prank = Some(prank);
164    Ok(Default::default())
165}
