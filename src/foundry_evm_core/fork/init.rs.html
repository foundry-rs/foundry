1use crate::utils::apply_chain_and_block_specific_env_changes;
2use alloy_consensus::BlockHeader;
3use alloy_primitives::{Address, U256};
4use alloy_provider::{network::BlockResponse, Network, Provider};
5use alloy_rpc_types::{BlockNumberOrTag, BlockTransactionsKind};
6use eyre::WrapErr;
7use foundry_common::NON_ARCHIVE_NODE_WARNING;
8use revm::primitives::{BlockEnv, CfgEnv, Env, TxEnv};
9
10/// Initializes a REVM block environment based on a forked
11/// ethereum provider.
12pub async fn environment<N: Network, P: Provider<N>>(
13    provider: &P,
14    memory_limit: u64,
15    gas_price: Option<u128>,
16    override_chain_id: Option<u64>,
17    pin_block: Option<u64>,
18    origin: Address,
19    disable_block_gas_limit: bool,
20) -> eyre::Result<(Env, N::BlockResponse)> {
21    let block_number = if let Some(pin_block) = pin_block {
22        pin_block
23    } else {
24        provider.get_block_number().await.wrap_err("failed to get latest block number")?
25    };
26    let (fork_gas_price, rpc_chain_id, block) = tokio::try_join!(
27        provider.get_gas_price(),
28        provider.get_chain_id(),
29        provider.get_block_by_number(
30            BlockNumberOrTag::Number(block_number),
31            BlockTransactionsKind::Hashes
32        )
33    )?;
34    let block = if let Some(block) = block {
35        block
36    } else {
37        if let Ok(latest_block) = provider.get_block_number().await {
38            // If the `eth_getBlockByNumber` call succeeds, but returns null instead of
39            // the block, and the block number is less than equal the latest block, then
40            // the user is forking from a non-archive node with an older block number.
41            if block_number <= latest_block {
42                error!("{NON_ARCHIVE_NODE_WARNING}");
43            }
44            eyre::bail!(
45                "failed to get block for block number: {block_number}; \
46                 latest block number: {latest_block}"
47            );
48        }
49        eyre::bail!("failed to get block for block number: {block_number}")
50    };
51
52    let mut cfg = CfgEnv::default();
53    cfg.chain_id = override_chain_id.unwrap_or(rpc_chain_id);
54    cfg.memory_limit = memory_limit;
55    cfg.limit_contract_code_size = Some(usize::MAX);
56    // EIP-3607 rejects transactions from senders with deployed code.
57    // If EIP-3607 is enabled it can cause issues during fuzz/invariant tests if the caller
58    // is a contract. So we disable the check by default.
59    cfg.disable_eip3607 = true;
60    cfg.disable_block_gas_limit = disable_block_gas_limit;
61
62    let mut env = Env {
63        cfg,
64        block: BlockEnv {
65            number: U256::from(block.header().number()),
66            timestamp: U256::from(block.header().timestamp()),
67            coinbase: block.header().beneficiary(),
68            difficulty: block.header().difficulty(),
69            prevrandao: block.header().mix_hash(),
70            basefee: U256::from(block.header().base_fee_per_gas().unwrap_or_default()),
71            gas_limit: U256::from(block.header().gas_limit()),
72            ..Default::default()
73        },
74        tx: TxEnv {
75            caller: origin,
76            gas_price: U256::from(gas_price.unwrap_or(fork_gas_price)),
77            chain_id: Some(override_chain_id.unwrap_or(rpc_chain_id)),
78            gas_limit: block.header().gas_limit() as u64,
79            ..Default::default()
80        },
81    };
82
83    apply_chain_and_block_specific_env_changes::<N>(&mut env, &block);
84
85    Ok((env, block))
86}
