29/// Depending on the configured chain id and block number this should apply any specific changes
30///
31/// - checks for prevrandao mixhash after merge
32/// - applies chain specifics: on Arbitrum `block.number` is the L1 block
33///
34/// Should be called with proper chain id (retrieved from provider if not provided).
35pub fn apply_chain_and_block_specific_env_changes<N: Network>(
36    env: &mut revm::primitives::Env,
37    block: &N::BlockResponse,
38) {
39    use NamedChain::*;
40    if let Ok(chain) = NamedChain::try_from(env.cfg.chain_id) {
41        let block_number = block.header().number();
42
43        match chain {
44            Mainnet => {
45                // after merge difficulty is supplanted with prevrandao EIP-4399
46                if block_number >= 15_537_351u64 {
47                    env.block.difficulty = env.block.prevrandao.unwrap_or_default().into();
48                }
49
50                return;
51            }
52            Moonbeam | Moonbase | Moonriver | MoonbeamDev => {
53                if env.block.prevrandao.is_none() {
54                    // <https://github.com/foundry-rs/foundry/issues/4232>
55                    env.block.prevrandao = Some(B256::random());
56                }
57            }
58            c if c.is_arbitrum() => {
59                // on arbitrum `block.number` is the L1 block which is included in the
60                // `l1BlockNumber` field
61                if let Some(l1_block_number) = block
62                    .other_fields()
63                    .and_then(|other| other.get("l1BlockNumber").cloned())
64                    .and_then(|l1_block_number| {
65                        serde_json::from_value::<U256>(l1_block_number).ok()
66                    })
67                {
68                    env.block.number = l1_block_number;
69                }
70            }
71            _ => {}
72        }
73    }
74
75    // if difficulty is `0` we assume it's past merge
76    if block.header().difficulty().is_zero() {
77        env.block.difficulty = env.block.prevrandao.unwrap_or_default().into();
78    }
79}

81/// Given an ABI and selector, it tries to find the respective function.
82pub fn get_function<'a>(
83    contract_name: &str,
84    selector: Selector,
85    abi: &'a JsonAbi,
86) -> eyre::Result<&'a Function> {
87    abi.functions()
88        .find(|func| func.selector() == selector)
89        .ok_or_else(|| eyre::eyre!("{contract_name} does not have the selector {selector}"))
90}

92/// Configures the env for the given RPC transaction.
93/// Accounts for an impersonated transaction by resetting the `env.tx.caller` field to `tx.from`.
94pub fn configure_tx_env(env: &mut revm::primitives::Env, tx: &Transaction<AnyTxEnvelope>) {
95    let impersonated_from = is_impersonated_tx(&tx.inner).then_some(tx.from);
96    if let AnyTxEnvelope::Ethereum(tx) = &tx.inner {
97        configure_tx_req_env(env, &tx.clone().into(), impersonated_from).expect("cannot fail");
98    }
99}

101/// Configures the env for the given RPC transaction request.
102/// `impersonated_from` is the address of the impersonated account. This helps account for an
103/// impersonated transaction by resetting the `env.tx.caller` field to `impersonated_from`.
104pub fn configure_tx_req_env(
105    env: &mut revm::primitives::Env,
106    tx: &TransactionRequest,
107    impersonated_from: Option<Address>,
108) -> eyre::Result<()> {
109    let TransactionRequest {
110        nonce,
111        from,
112        to,
113        value,
114        gas_price,
115        gas,
116        max_fee_per_gas,
117        max_priority_fee_per_gas,
118        max_fee_per_blob_gas,
119        ref input,
120        chain_id,
121        ref blob_versioned_hashes,
122        ref access_list,
123        transaction_type: _,
124        ref authorization_list,
125        sidecar: _,
126    } = *tx;
127
128    // If no `to` field then set create kind: https://eips.ethereum.org/EIPS/eip-2470#deployment-transaction
129    env.tx.transact_to = to.unwrap_or(TxKind::Create);
130    // If the transaction is impersonated, we need to set the caller to the from
131    // address Ref: https://github.com/foundry-rs/foundry/issues/9541
132    env.tx.caller =
133        impersonated_from.unwrap_or(from.ok_or_else(|| eyre::eyre!("missing `from` field"))?);
134    env.tx.gas_limit = gas.ok_or_else(|| eyre::eyre!("missing `gas` field"))?;
135    env.tx.nonce = nonce;
136    env.tx.value = value.unwrap_or_default();
137    env.tx.data = input.input().cloned().unwrap_or_default();
138    env.tx.chain_id = chain_id;
139
140    // Type 1, EIP-2930
141    env.tx.access_list = access_list.clone().unwrap_or_default().0.into_iter().collect();
142
143    // Type 2, EIP-1559
144    env.tx.gas_price = U256::from(gas_price.or(max_fee_per_gas).unwrap_or_default());
145    env.tx.gas_priority_fee = max_priority_fee_per_gas.map(U256::from);
146
147    // Type 3, EIP-4844
148    env.tx.blob_hashes = blob_versioned_hashes.clone().unwrap_or_default();
149    env.tx.max_fee_per_blob_gas = max_fee_per_blob_gas.map(U256::from);
150
151    // Type 4, EIP-7702
152    if let Some(authorization_list) = authorization_list {
153        env.tx.authorization_list =
154            Some(revm::primitives::AuthorizationList::Signed(authorization_list.clone()));
155    }
156
157    Ok(())
158}

160/// Get the gas used, accounting for refunds
161pub fn gas_used(spec: SpecId, spent: u64, refunded: u64) -> u64 {
162    let refund_quotient = if SpecId::enabled(spec, SpecId::LONDON) { 5 } else { 2 };
163    spent - (refunded).min(spent / refund_quotient)
164}

166fn get_create2_factory_call_inputs(
167    salt: U256,
168    inputs: CreateInputs,
169    deployer: Address,
170) -> CallInputs {
171    let calldata = [&salt.to_be_bytes::<32>()[..], &inputs.init_code[..]].concat();
172    CallInputs {
173        caller: inputs.caller,
174        bytecode_address: deployer,
175        target_address: deployer,
176        scheme: CallScheme::Call,
177        value: CallValue::Transfer(inputs.value),
178        input: calldata.into(),
179        gas_limit: inputs.gas_limit,
180        is_static: false,
181        return_memory_offset: 0..0,
182        is_eof: false,
183    }
184}

186/// Used for routing certain CREATE2 invocations through CREATE2_DEPLOYER.
187///
188/// Overrides create hook with CALL frame if [InspectorExt::should_use_create2_factory] returns
189/// true. Keeps track of overridden frames and handles outcome in the overridden insert_call_outcome
190/// hook by inserting decoded address directly into interpreter.
191///
192/// Should be installed after [revm::inspector_handle_register] and before any other registers.
193pub fn create2_handler_register<I: InspectorExt>(
194    handler: &mut EvmHandler<'_, I, &mut dyn DatabaseExt>,
195) {
196    let create2_overrides = Rc::<RefCell<Vec<_>>>::new(RefCell::new(Vec::new()));
197
198    let create2_overrides_inner = create2_overrides.clone();
199    let old_handle = handler.execution.create.clone();
200    handler.execution.create =
201        Arc::new(move |ctx, mut inputs| -> Result<FrameOrResult, EVMError<DatabaseError>> {
202            let CreateScheme::Create2 { salt } = inputs.scheme else {
203                return old_handle(ctx, inputs);
204            };
205            if !ctx.external.should_use_create2_factory(&mut ctx.evm, &mut inputs) {
206                return old_handle(ctx, inputs);
207            }
208
209            let gas_limit = inputs.gas_limit;
210
211            // Get CREATE2 deployer.
212            let create2_deployer = ctx.external.create2_deployer();
213            // Generate call inputs for CREATE2 factory.
214            let mut call_inputs = get_create2_factory_call_inputs(salt, *inputs, create2_deployer);
215
216            // Call inspector to change input or return outcome.
217            let outcome = ctx.external.call(&mut ctx.evm, &mut call_inputs);
218
219            // Push data about current override to the stack.
220            create2_overrides_inner
221                .borrow_mut()
222                .push((ctx.evm.journaled_state.depth(), call_inputs.clone()));
223
224            // Sanity check that CREATE2 deployer exists.
225            let code_hash = ctx.evm.load_account(create2_deployer)?.info.code_hash;
226            if code_hash == KECCAK_EMPTY {
227                return Ok(FrameOrResult::Result(FrameResult::Call(CallOutcome {
228                    result: InterpreterResult {
229                        result: InstructionResult::Revert,
230                        output: format!("missing CREATE2 deployer: {create2_deployer}").into(),
231                        gas: Gas::new(gas_limit),
232                    },
233                    memory_offset: 0..0,
234                })))
235            } else if code_hash != DEFAULT_CREATE2_DEPLOYER_CODEHASH {
236                return Ok(FrameOrResult::Result(FrameResult::Call(CallOutcome {
237                    result: InterpreterResult {
238                        result: InstructionResult::Revert,
239                        output: "invalid CREATE2 deployer bytecode".into(),
240                        gas: Gas::new(gas_limit),
241                    },
242                    memory_offset: 0..0,
243                })))
244            }
245
246            // Handle potential inspector override.
247            if let Some(outcome) = outcome {
248                return Ok(FrameOrResult::Result(FrameResult::Call(outcome)));
249            }
250
251            // Create CALL frame for CREATE2 factory invocation.
252            let mut frame_or_result = ctx.evm.make_call_frame(&call_inputs);
253
254            if let Ok(FrameOrResult::Frame(frame)) = &mut frame_or_result {
255                ctx.external
256                    .initialize_interp(&mut frame.frame_data_mut().interpreter, &mut ctx.evm)
257            }
258            frame_or_result
259        });
260
261    let create2_overrides_inner = create2_overrides;
262    let old_handle = handler.execution.insert_call_outcome.clone();
263    handler.execution.insert_call_outcome =
264        Arc::new(move |ctx, frame, shared_memory, mut outcome| {
265            // If we are on the depth of the latest override, handle the outcome.
266            if create2_overrides_inner
267                .borrow()
268                .last()
269                .is_some_and(|(depth, _)| *depth == ctx.evm.journaled_state.depth())
270            {
271                let (_, call_inputs) = create2_overrides_inner.borrow_mut().pop().unwrap();
272                outcome = ctx.external.call_end(&mut ctx.evm, &call_inputs, outcome);
273
274                // Decode address from output.
275                let address = match outcome.instruction_result() {
276                    return_ok!() => Address::try_from(outcome.output().as_ref())
277                        .map_err(|_| {
278                            outcome.result = InterpreterResult {
279                                result: InstructionResult::Revert,
280                                output: "invalid CREATE2 factory output".into(),
281                                gas: Gas::new(call_inputs.gas_limit),
282                            };
283                        })
284                        .ok(),
285                    _ => None,
286                };
287                frame
288                    .frame_data_mut()
289                    .interpreter
290                    .insert_create_outcome(CreateOutcome { address, result: outcome.result });
291
292                Ok(())
293            } else {
294                old_handle(ctx, frame, shared_memory, outcome)
295            }
296        });
297}

299/// Adds Odyssey P256 precompile to the list of loaded precompiles.
300pub fn odyssey_handler_register<EXT, DB: revm::Database>(handler: &mut EvmHandler<'_, EXT, DB>) {
301    let prev = handler.pre_execution.load_precompiles.clone();
302    handler.pre_execution.load_precompiles = Arc::new(move || {
303        let mut loaded_precompiles = prev();
304
305        loaded_precompiles.extend([ODYSSEY_P256, P256VERIFY]);
306
307        loaded_precompiles
308    });
309}

311/// Creates a new EVM with the given inspector.
312pub fn new_evm_with_inspector<'evm, 'i, 'db, I: InspectorExt + ?Sized>(
313    db: &'db mut dyn DatabaseExt,
314    env: revm::primitives::EnvWithHandlerCfg,
315    inspector: &'i mut I,
316) -> revm::Evm<'evm, &'i mut I, &'db mut dyn DatabaseExt> {
317    let revm::primitives::EnvWithHandlerCfg { env, handler_cfg } = env;
318
319    // NOTE: We could use `revm::Evm::builder()` here, but on the current patch it has some
320    // performance issues.
321    /*
322    revm::Evm::builder()
323        .with_db(db)
324        .with_env(env)
325        .with_external_context(inspector)
326        .with_handler_cfg(handler_cfg)
327        .append_handler_register(revm::inspector_handle_register)
328        .append_handler_register(create2_handler_register)
329        .build()
330    */
331
332    let mut handler = revm::Handler::new(handler_cfg);
333    handler.append_handler_register_plain(revm::inspector_handle_register);
334    if inspector.is_odyssey() {
335        handler.append_handler_register_plain(odyssey_handler_register);
336    }
337    handler.append_handler_register_plain(create2_handler_register);
338
339    let context = revm::Context::new(revm::EvmContext::new_with_env(db, env), inspector);
340
341    revm::Evm::new(context, handler)
342}

344pub fn new_evm_with_existing_context<'a>(
345    inner: revm::InnerEvmContext<&'a mut dyn DatabaseExt>,
346    inspector: &'a mut dyn InspectorExt,
347) -> revm::Evm<'a, &'a mut dyn InspectorExt, &'a mut dyn DatabaseExt> {
348    let handler_cfg = HandlerCfg::new(inner.spec_id());
349
350    let mut handler = revm::Handler::new(handler_cfg);
351    handler.append_handler_register_plain(revm::inspector_handle_register);
352    if inspector.is_odyssey() {
353        handler.append_handler_register_plain(odyssey_handler_register);
354    }
355    handler.append_handler_register_plain(create2_handler_register);
356
357    let context =
358        revm::Context::new(revm::EvmContext { inner, precompiles: Default::default() }, inspector);
359    revm::Evm::new(context, handler)
